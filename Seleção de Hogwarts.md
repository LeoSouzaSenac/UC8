    private void chapeuBtnActionPerformed(java.awt.event.ActionEvent evt) {                                          
        int linhaSelecionada = tabelaEspera.getSelectedRow();
        if (linhaSelecionada != -1) {
            String nome = (String) modeloEspera.getValueAt(linhaSelecionada, 0);
            String personalidade = (String) modeloEspera.getValueAt(linhaSelecionada, 1);

                        
            modeloEspera.removeRow(linhaSelecionada);

            switch (personalidade.toLowerCase()) {
                case "coragem":
                    modeloGrifinoria.addRow(new Object[]{nome, personalidade});
                    break;
                case "ast√∫cia":
                    modeloSonserina.addRow(new Object[]{nome, personalidade});
                    break;
                case "lealdade":
                    modeloLufaLufa.addRow(new Object[]{nome, personalidade});
                    break;
                case "intelig√™ncia":
                    modeloCorvinal.addRow(new Object[]{nome, personalidade});
                    break;
                default:
                    JOptionPane.showMessageDialog(null, "Casa ainda n√£o implementada para esta personalidade.");
            }

            atualizarLabels(); // Atualiza os contadores das casas
        }
    }                                         

    private void expulsarBtnActionPerformed(java.awt.event.ActionEvent evt) {                                            
        if (tabelaGrifinoria.getSelectedRow() != -1) {
            removerAlunoDaCasa(tabelaGrifinoria, modeloGrifinoria);
        } else if (tabelaSonserina.getSelectedRow() != -1) {
            removerAlunoDaCasa(tabelaSonserina, modeloSonserina);
        } else if (tabelaCorvinal.getSelectedRow() != -1) {
            removerAlunoDaCasa(tabelaCorvinal, modeloCorvinal);
        } else if (tabelaLufaLufa.getSelectedRow() != -1) {
            removerAlunoDaCasa(tabelaLufaLufa, modeloLufaLufa);
        } else {
            JOptionPane.showMessageDialog(this, "Selecione um aluno em uma das casas.");
        }

        atualizarLabels(); // Atualiza os contadores das casas
    }

    private void removerAlunoDaCasa(JTable tabelaCasa, DefaultTableModel modeloCasa) {
        int selectedRow = tabelaCasa.getSelectedRow();

        if (selectedRow == -1) {
            JOptionPane.showMessageDialog(this, "Selecione um aluno em uma das casas.");
            return;
        }

        String nome = modeloCasa.getValueAt(selectedRow, 0).toString();
        String personalidade = modeloCasa.getValueAt(selectedRow, 1).toString();

        // Adiciona de volta √† tabela de espera
        modeloEspera.addRow(new Object[]{nome, personalidade});
        modeloCasa.removeRow(selectedRow);

        atualizarLabels(); // Atualiza os contadores das casas
    }
